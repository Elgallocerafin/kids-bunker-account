#include "hookapi.h"

// ============================================================================
// CONFIGURACIÓN DE TUTORES (Introduce tus 20 bytes HEX reales aquí)
// ============================================================================
#define TUTOR_1 { 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07, 0x08, 0x09, 0x0A, 0x0B, 0x0C, 0x0D, 0x0E, 0x0F, 0x10, 0x11, 0x12, 0x13, 0x14 }
#define TUTOR_2 { 0x1A, 0x2B, 0x3C, 0x4D, 0x5E, 0x6F, 0x7A, 0x8B, 0x9C, 0x0D, 0x1E, 0x2F, 0x3A, 0x4B, 0x5C, 0x6D, 0x7E, 0x8F, 0x9A, 0x0B }

// Marca de tiempo de simulación o real para la apertura
#define FECHA_MAYORIA_EDAD 5000

int64_t hook(uint32_t reserved) {
    
    _g(1, 1); 

    int64_t txn_type = otxn_type();

    // Si es la instalación del Hook (SetHook), aprobamos de inmediato sin hacer nada
    if (txn_type != ttPAYMENT) {
        accept(0, 0, 0); 
    }

    // Recoger datos clave de la transacción actual
    uint8_t emisor[20];
    otxn_field((uint32_t)emisor, 20, sfAccount);

    uint8_t cuenta_niña[20];
    hook_account((uint32_t)cuenta_niña, 20);

    // Identificar si quien opera es la dueña de la cuenta
    int es_la_niña = BUFFER_EQUAL_20(emisor, cuenta_niña);
    uint8_t llave_contador[1] = {0x55};

    // ========================================================================
    // RUTA A: ENTRADA DE FONDOS Y REGISTRO DE VOTOS (Cualquier emisor externo)
    // ========================================================================
    if (!es_la_niña) {
        uint8_t memo_raw[32]; 
        int64_t memo_len = otxn_field((uint32_t)memo_raw, 32, sfMemos);

        int tiene_voto_memo = 0;
        
        // Escáner plano original de 25 posiciones
        if (memo_len >= 4) {
            if ((memo_raw[0] == 'V' && memo_raw[1] == 'O' && memo_raw[2] == 'T' && memo_raw[3] == 'O') ||
                (memo_raw[1] == 'V' && memo_raw[2] == 'O' && memo_raw[3] == 'T' && memo_raw[4] == 'O') ||
                (memo_raw[2] == 'V' && memo_raw[3] == 'O' && memo_raw[4] == 'T' && memo_raw[5] == 'O') ||
                (memo_raw[3] == 'V' && memo_raw[4] == 'O' && memo_raw[5] == 'T' && memo_raw[6] == 'O') ||
                (memo_raw[4] == 'V' && memo_raw[5] == 'O' && memo_raw[6] == 'T' && memo_raw[7] == 'O') ||
                (memo_raw[5] == 'V' && memo_raw[6] == 'O' && memo_raw[7] == 'T' && memo_raw[8] == 'O') ||
                (memo_raw[6] == 'V' && memo_raw[7] == 'O' && memo_raw[8] == 'T' && memo_raw[9] == 'O') ||
                (memo_raw[7] == 'V' && memo_raw[8] == 'O' && memo_raw[9] == 'T' && memo_raw[10] == 'O') ||
                (memo_raw[8] == 'V' && memo_raw[9] == 'O' && memo_raw[10] == 'T' && memo_raw[11] == 'O') ||
                (memo_raw[9] == 'V' && memo_raw[10] == 'O' && memo_raw[11] == 'T' && memo_raw[12] == 'O') ||
                (memo_raw[10] == 'V' && memo_raw[11] == 'O' && memo_raw[12] == 'T' && memo_raw[13] == 'O') ||
                (memo_raw[11] == 'V' && memo_raw[12] == 'O' && memo_raw[13] == 'T' && memo_raw[14] == 'O') ||
                (memo_raw[12] == 'V' && memo_raw[13] == 'O' && memo_raw[14] == 'T' && memo_raw[15] == 'O') ||
                (memo_raw[13] == 'V' && memo_raw[14] == 'O' && memo_raw[15] == 'T' && memo_raw[16] == 'O') ||
                (memo_raw[14] == 'V' && memo_raw[15] == 'O' && memo_raw[16] == 'T' && memo_raw[17] == 'O') ||
                (memo_raw[15] == 'V' && memo_raw[16] == 'O' && memo_raw[17] == 'T' && memo_raw[18] == 'O') ||
                (memo_raw[16] == 'V' && memo_raw[17] == 'O' && memo_raw[18] == 'T' && memo_raw[19] == 'O') ||
                (memo_raw[17] == 'V' && memo_raw[18] == 'O' && memo_raw[19] == 'T' && memo_raw[20] == 'O') ||
                (memo_raw[18] == 'V' && memo_raw[19] == 'O' && memo_raw[20] == 'T' && memo_raw[21] == 'O') ||
                (memo_raw[19] == 'V' && memo_raw[20] == 'O' && memo_raw[21] == 'T' && memo_raw[22] == 'O') ||
                (memo_raw[20] == 'V' && memo_raw[21] == 'O' && memo_raw[22] == 'T' && memo_raw[23] == 'O') ||
                (memo_raw[21] == 'V' && memo_raw[22] == 'O' && memo_raw[23] == 'T' && memo_raw[24] == 'O') ||
                (memo_raw[22] == 'V' && memo_raw[23] == 'O' && memo_raw[24] == 'T' && memo_raw[25] == 'O') ||
                (memo_raw[23] == 'V' && memo_raw[24] == 'O' && memo_raw[25] == 'T' && memo_raw[26] == 'O') ||
                (memo_raw[24] == 'V' && memo_raw[25] == 'O' && memo_raw[26] == 'T' && memo_raw[27] == 'O')) {
                tiene_voto_memo = 1;
            }
        }

        if (tiene_voto_memo) {
            uint8_t t1[20] = TUTOR_1;
            uint8_t t2[20] = TUTOR_2;

            int coincide_t1 = BUFFER_EQUAL_20(emisor, t1);
            int coincide_t2 = BUFFER_EQUAL_20(emisor, t2);

            if (!coincide_t1 && !coincide_t2) {
                rollback((uint32_t)"Voto no autorizado", 18, 1);
            }

            uint8_t votos_actuales = 0;
            state((uint32_t)&votos_actuales, 1, (uint32_t)llave_contador, 1);
            votos_actuales++;
            state_set((uint32_t)&votos_actuales, 1, (uint32_t)llave_contador, 1);
            
            trace((uint32_t)"[VOTO] Registrado en el State.", 30, 0, 0, 1);
        }

        // Cualquier pago entrante se acepta sin trabas
        accept(0, 0, 0);
    }

    // ========================================================================
    // RUTA B: CONTROL DE SALIDA Y AUTO-DESTRUCCIÓN (Activado por la Niña)
    // ========================================================================
    if (ledger_last_time() < FECHA_MAYORIA_EDAD) {
        rollback((uint32_t)"Bunker bloqueado: Menor de edad", 31, 2);
    }

    uint8_t votos_totales = 0;
    state((uint32_t)&votos_totales, 1, (uint32_t)llave_contador, 1);

    if (votos_totales < 2) {
        rollback((uint32_t)"Bunker bloqueado: Faltan votos", 30, 3);
    }

    // ------------------------------------------------------------------------
    // ¡CONSENSO ALCANZADO! Emitimos una transacción para BORRAR el Hook
    // ------------------------------------------------------------------------
    trace((uint32_t)"[LIBERACION] Ejecutando autodestruccion del Hook...", 51, 0, 0, 1);

    // Estructura de bytes para una transacción SetHook de borrado total (Vacía)
    uint8_t tx_borrado[48] = {
        0x12, 0x00, 0x16,                  // ttSET_HOOK (Tipo de transacción: 22)
        0x24, 0x00, 0x00, 0x00, 0x00,      // sfSequence
        0x25, 0x00, 0x00, 0x00, 0x00,      // sfFee
        0x65,                              // sfSetHook -> Comienza el Array
        0xE1,                              // STObject de instalación
        0x76, 0x00, 0x00, 0x00, 0x00,      // sfHookHash vacío -> Indica eliminación
        0xE1,                              // Fin del STObject
        0xF1                               // Fin del Array sfSetHook
    };

    // Buffer para recibir el Hash de la transacción emitida (32 bytes)
    uint8_t tx_id[32];

    // Llamada corregida con los 4 argumentos reglamentarios de Xahau
    emit((uint32_t)tx_borrado, sizeof(tx_borrado), (uint32_t)tx_id, 32);

    // Borramos la clave del contador para no dejar "polvo" informático en el ledger
    state_set(0, 0, (uint32_t)llave_contador, 1);

    trace((uint32_t)"[EXITO] Transaccion de borrado emitida. Liberando...", 52, 0, 0, 1);
    
    accept(0, 0, 0);

    return 0;
}
